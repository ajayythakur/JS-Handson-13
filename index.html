Qn 3. Create a promise function accepting a argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject state and catch the error and print Promise Rejected.

Ans - 

        function abc(a){
          let z=new Promise((res,rej)=>{
            if(a=="Yes"){
              res("Promise Resolved")
            }
            else{
              rej(new Error("Promise Rejected"));
            }
          }).then(function(data){
            console.log(data);
          }).catch(function(err){
            console.log(err);
